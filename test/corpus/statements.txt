  class(*), intent(in) :: number

  select type(number)
  type is (integer)
    write(*, '("<", i0, ">")') number
  type is (real(kind(1.0)))
    write(*, '("(", f8.5, ")")') number
  type is (real(kind(1.0d0)))
    write(*, '("{", f13.8, "}")') number
  type is (complex)
    write(*, '("[", f13.8, " + ", f13.8 "i]")') number
  class is (custom_type)
    write(*, '("#", f8.5, "#")') number%value

    write(*, '("~Not a number~")')

    rank (1)
      write(*, *) "rank 1"
    rank (2)
      write(*, *) "rank 2"
    rank default
